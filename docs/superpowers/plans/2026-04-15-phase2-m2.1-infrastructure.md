> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把 Phase 1 的 JSONL 存储、手写 Hook 协议、子串 Matcher 替换成 2026 最成熟的开源基础设施（`node:sqlite` / `@anthropic-ai/claude-agent-sdk` / `web-tree-sitter`），把 Phase 1 32 条知识干净迁移到 v2 schema（全部回到 experimental tier），为 M2.2 Calibrator v2 + 后续所有 milestone 铺路。

**Tech Stack:** `node:sqlite` (Node ≥ 22)、`sqlite-vec`（扩展加载）、`web-tree-sitter` (WASM)、`@anthropic-ai/claude-agent-sdk-typescript`、`zod` v4、`mitt`、`cli-table3`、`picocolors`、`js-tiktoken`。Phase 1 已有 `tsx` / `vitest` / `tsup` 不动。

```typescript
import { DatabaseSync } from "node:sqlite";

/** 所有 DDL 集中在这里，首次打开 DB 时幂等执行一次。 */
export const INIT_SQL = `
-- 知识主表
CREATE TABLE IF NOT EXISTS knowledge (
  id TEXT PRIMARY KEY,
  scope_level TEXT NOT NULL CHECK(scope_level IN ('personal','team','global')),
  scope_project TEXT,
  scope_paths TEXT,                -- JSON array
  scope_file_types TEXT,           -- JSON array
  scope_branches TEXT,             -- JSON array
  category TEXT NOT NULL,          -- C/E/S/K/W
  tags TEXT,                       -- JSON array
  type TEXT NOT NULL,              -- avoidance / practice / wiki / success
  nature TEXT NOT NULL,            -- objective / subjective
  trigger TEXT NOT NULL,
  wrong_pattern TEXT DEFAULT '',
  correct_pattern TEXT NOT NULL,
  correct_pattern_code_example TEXT,
  correct_pattern_import_path TEXT,
  correct_pattern_tldr TEXT,
  reasoning TEXT,
  when_expression TEXT,            -- CEL, 可空
  confidence REAL NOT NULL DEFAULT 0,
  demerit REAL NOT NULL DEFAULT 0,
  demerit_last_updated TEXT,
  current_tier TEXT NOT NULL DEFAULT 'experimental'
    CHECK(current_tier IN ('experimental','probation','stable','canonical','enforced','dormant')),
  max_tier_ever TEXT NOT NULL DEFAULT 'experimental',
  tier_entered_at TEXT NOT NULL,
  enforcement TEXT NOT NULL DEFAULT 'passive',
  status TEXT NOT NULL DEFAULT 'active'
    CHECK(status IN ('active','conflict','stale','archived','dormant')),
  hit_count INTEGER NOT NULL DEFAULT 0,
  success_count INTEGER NOT NULL DEFAULT 0,
  override_count INTEGER NOT NULL DEFAULT 0,
  resurrect_count INTEGER NOT NULL DEFAULT 0,
  evidence TEXT,                   -- JSON
  source TEXT NOT NULL,            -- preset/imported/accumulated/team-shared/internet/detector/insights/npm_audit/pr/git_log/ci
  conflict_with TEXT,              -- JSON array
  created_at TEXT NOT NULL,
  last_hit_at TEXT,
  last_validated_at TEXT
);

CREATE INDEX IF NOT EXISTS idx_knowledge_tier ON knowledge(current_tier);
CREATE INDEX IF NOT EXISTS idx_knowledge_scope ON knowledge(scope_level, scope_project);
CREATE INDEX IF NOT EXISTS idx_knowledge_status ON knowledge(status);

-- 观察表（Calibrator 用，v2 新增）
CREATE TABLE IF NOT EXISTS observations (
  id TEXT PRIMARY KEY,
  knowledge_id TEXT NOT NULL,
  timestamp TEXT NOT NULL,
  outcome TEXT NOT NULL CHECK(outcome IN ('success','failure')),
  source_event TEXT,
  tool_use_id TEXT,
  FOREIGN KEY(knowledge_id) REFERENCES knowledge(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_obs_knowledge ON observations(knowledge_id, timestamp DESC);

-- 事件表（替代 JsonlEventLog，全历史 append-only）
CREATE TABLE IF NOT EXISTS events (
  id TEXT PRIMARY KEY,
  kind TEXT NOT NULL,
  knowledge_id TEXT,
  tool_use_id TEXT,
  timestamp TEXT NOT NULL,
  payload TEXT                     -- JSON
);

CREATE INDEX IF NOT EXISTS idx_events_kind ON events(kind, timestamp DESC);
CREATE INDEX IF NOT EXISTS idx_events_knowledge ON events(knowledge_id);

-- Wiki 专用元数据表（SP-3 用，M2.1 先建表，实际填充在 M2.6）
CREATE TABLE IF NOT EXISTS wiki_meta (
  knowledge_id TEXT PRIMARY KEY,
  source_url TEXT,
  source_type TEXT,
  published_at TEXT,
  tldr TEXT,
  keywords TEXT,
  user_thumbs_down INTEGER DEFAULT 0,
  inline_injection_count INTEGER DEFAULT 0,
  FOREIGN KEY(knowledge_id) REFERENCES knowledge(id) ON DELETE CASCADE
);

-- schema 版本表（后续 migration 用）
CREATE TABLE IF NOT EXISTS schema_version (
  version INTEGER PRIMARY KEY,
  applied_at TEXT NOT NULL
);
INSERT OR IGNORE INTO schema_version(version, applied_at) VALUES (1, datetime('now'));
`;

export const CURRENT_SCHEMA_VERSION = 1;

/**
 * 打开/创建 SQLite DB 并确保 schema 初始化。
 * 幂等——重复调用无副作用。
 */
export function openDb(path: string): DatabaseSync {
  const db = new DatabaseSync(path);
  db.exec("PRAGMA journal_mode = WAL;");
  db.exec("PRAGMA foreign_keys = ON;");
  db.exec(INIT_SQL);
  return db;
}

/** 关闭连接（测试 cleanup 用）。 */
export function closeDb(db: DatabaseSync): void {
  db.close();
}
```

Create `packages/adapters/src/storage/sqlite/sqlite-knowledge-store.ts`：

```typescript
import type { DatabaseSync } from "node:sqlite";
import type { KnowledgeEntry, Scope } from "@teamagent/types";

/**
 * KnowledgeStore 的 SQLite 实现。v2 的主存储。
 * 专注 CRUD；列表/查询/更新在后续 task 加。
 */
export class SqliteKnowledgeStore {
  private readonly db: DatabaseSync;

  constructor(db: DatabaseSync) {
    this.db = db;
  }

  add(entry: KnowledgeEntry): void {
    const stmt = this.db.prepare(`
      INSERT INTO knowledge (
        id, scope_level, scope_project, scope_paths, scope_file_types, scope_branches,
        category, tags, type, nature,
        trigger, wrong_pattern, correct_pattern,
        correct_pattern_code_example, correct_pattern_import_path, correct_pattern_tldr,
        reasoning, when_expression,
        confidence, demerit, demerit_last_updated,
        current_tier, max_tier_ever, tier_entered_at,
        enforcement, status,
        hit_count, success_count, override_count, resurrect_count,
        evidence, source, conflict_with,
        created_at, last_hit_at, last_validated_at
      ) VALUES (
        @id, @scope_level, @scope_project, @scope_paths, @scope_file_types, @scope_branches,
        @category, @tags, @type, @nature,
        @trigger, @wrong_pattern, @correct_pattern,
        @correct_pattern_code_example, @correct_pattern_import_path, @correct_pattern_tldr,
        @reasoning, @when_expression,
        @confidence, @demerit, @demerit_last_updated,
        @current_tier, @max_tier_ever, @tier_entered_at,
        @enforcement, @status,
        @hit_count, @success_count, @override_count, @resurrect_count,
        @evidence, @source, @conflict_with,
        @created_at, @last_hit_at, @last_validated_at
      )
    `);
    stmt.run(this.serialize(entry));
  }

  getById(id: string): KnowledgeEntry | undefined {
    const row = this.db.prepare("SELECT * FROM knowledge WHERE id = ?").get(id) as any;
    if (!row) return undefined;
    return this.deserialize(row);
  }

  close(): void {
    this.db.close();
  }

  // ---- 序列化 helpers ----

  private serialize(e: KnowledgeEntry): Record<string, unknown> {
    return {
      id: e.id,
      scope_level: e.scope.level,
      scope_project: e.scope.project ?? null,
      scope_paths: jsonOrNull(e.scope.paths),
      scope_file_types: jsonOrNull(e.scope.file_types),
      scope_branches: jsonOrNull(e.scope.branches),
      category: e.category,
      tags: JSON.stringify(e.tags ?? []),
      type: e.type,
      nature: e.nature,
      trigger: e.trigger,
      wrong_pattern: e.wrong_pattern ?? "",
      correct_pattern: e.correct_pattern,
      correct_pattern_code_example: (e as any).correct_pattern_code_example ?? null,
      correct_pattern_import_path: (e as any).correct_pattern_import_path ?? null,
      correct_pattern_tldr: (e as any).correct_pattern_tldr ?? null,
      reasoning: e.reasoning,
      when_expression: (e as any).when_expression ?? null,
      confidence: e.confidence,
      demerit: (e as any).demerit ?? 0,
      demerit_last_updated: (e as any).demerit_last_updated ?? null,
      current_tier: (e as any).current_tier ?? "experimental",
      max_tier_ever: (e as any).max_tier_ever ?? "experimental",
      tier_entered_at: (e as any).tier_entered_at ?? e.created_at,
      enforcement: e.enforcement,
      status: e.status,
      hit_count: e.hit_count,
      success_count: e.success_count,
      override_count: e.override_count,
      resurrect_count: (e as any).resurrect_count ?? 0,
      evidence: JSON.stringify(e.evidence),
      source: e.source,
      conflict_with: JSON.stringify(e.conflict_with ?? []),
      created_at: e.created_at,
      last_hit_at: e.last_hit_at ?? "",
      last_validated_at: e.last_validated_at ?? "",
    };
  }

  private deserialize(row: any): KnowledgeEntry {
    const scope: Scope = {
      level: row.scope_level,
      project: row.scope_project ?? undefined,
      paths: parseJsonArray(row.scope_paths),
      file_types: parseJsonArray(row.scope_file_types),
      branches: parseJsonArray(row.scope_branches),
    };
    return {
      id: row.id,
      scope,
      category: row.category,
      tags: parseJsonArray(row.tags) ?? [],
      type: row.type,
      nature: row.nature,
      trigger: row.trigger,
      wrong_pattern: row.wrong_pattern ?? "",
      correct_pattern: row.correct_pattern,
      reasoning: row.reasoning,
      confidence: row.confidence,
      enforcement: row.enforcement,
      status: row.status,
      hit_count: row.hit_count,
      success_count: row.success_count,
      override_count: row.override_count,
      evidence: row.evidence ? JSON.parse(row.evidence) : { success_sessions: 0, success_users: 0, correction_sessions: 0 },
      created_at: row.created_at,
      last_hit_at: row.last_hit_at ?? "",
      last_validated_at: row.last_validated_at ?? "",
      source: row.source,
      conflict_with: parseJsonArray(row.conflict_with) ?? [],
    };
  }
}

function jsonOrNull(v: unknown): string | null {
  return v === undefined || v === null ? null : JSON.stringify(v);
}

function parseJsonArray(v: string | null | undefined): string[] | undefined {
  if (!v) return undefined;
  try {
    const parsed = JSON.parse(v);
    return Array.isArray(parsed) ? parsed : undefined;
  } catch {
    return undefined;
  }
}
```

```typescript
import type { DatabaseSync } from "node:sqlite";
import type { PersistedEvent } from "@teamagent/types";

const CORE_KEYS = new Set(["id", "kind", "knowledge_id", "tool_use_id", "timestamp", "schema_version"]);

export class SqliteEventLog {
  private readonly db: DatabaseSync;

  constructor(db: DatabaseSync) {
    this.db = db;
  }

  append(e: PersistedEvent): void {
    // 核心字段单独列；其它字段（confidence_before 等）塞进 payload JSON
    const payload: Record<string, unknown> = {};
    for (const [k, v] of Object.entries(e)) {
      if (!CORE_KEYS.has(k)) payload[k] = v;
    }

    this.db.prepare(`
      INSERT INTO events (id, kind, knowledge_id, tool_use_id, timestamp, payload)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(
      e.id,
      e.kind,
      (e as any).knowledge_id ?? null,
      (e as any).tool_use_id ?? null,
      e.timestamp,
      Object.keys(payload).length ? JSON.stringify(payload) : null,
    );
  }

  readAll(): PersistedEvent[] {
    const rows = this.db.prepare("SELECT * FROM events ORDER BY timestamp ASC").all() as any[];
    return rows.map(this.hydrate);
  }

  readByKind(kind: string): PersistedEvent[] {
    const rows = this.db
      .prepare("SELECT * FROM events WHERE kind = ? ORDER BY timestamp ASC")
      .all(kind) as any[];
    return rows.map(this.hydrate);
  }

  readLast(n: number): PersistedEvent[] {
    const rows = this.db
      .prepare("SELECT * FROM events ORDER BY timestamp DESC LIMIT ?")
      .all(n) as any[];
    return rows.map(this.hydrate);
  }

  close(): void {
    this.db.close();
  }

  private hydrate = (row: any): PersistedEvent => {
    const extra = row.payload ? JSON.parse(row.payload) : {};
    return {
      id: row.id,
      kind: row.kind,
      knowledge_id: row.knowledge_id ?? undefined,
      tool_use_id: row.tool_use_id ?? undefined,
      timestamp: row.timestamp,
      schema_version: 1,
      ...extra,
    };
  };
}
```

```typescript
import type { DatabaseSync } from "node:sqlite";

export interface Observation {
  id: string;
  knowledge_id: string;
  timestamp: string;
  outcome: "success" | "failure";
  source_event?: string;
  tool_use_id?: string;
}

export class SqliteObservations {
  private readonly db: DatabaseSync;

  constructor(db: DatabaseSync) {
    this.db = db;
  }

  add(o: Observation): void {
    this.db.prepare(`
      INSERT INTO observations (id, knowledge_id, timestamp, outcome, source_event, tool_use_id)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(o.id, o.knowledge_id, o.timestamp, o.outcome, o.source_event ?? null, o.tool_use_id ?? null);
  }

  listForKnowledge(knowledge_id: string): Observation[] {
    const rows = this.db.prepare(`
      SELECT id, knowledge_id, timestamp, outcome, source_event, tool_use_id
      FROM observations WHERE knowledge_id = ?
      ORDER BY timestamp DESC
    `).all(knowledge_id) as Observation[];
    return rows;
  }

  countByOutcome(knowledge_id: string, outcome: "success" | "failure"): number {
    const row = this.db.prepare(`
      SELECT COUNT(*) as n FROM observations
      WHERE knowledge_id = ? AND outcome = ?
    `).get(knowledge_id, outcome) as { n: number };
    return row.n;
  }

  close(): void {
    this.db.close();
  }
}
```

```typescript
import { openDb } from "./schema.js";
import { SqliteKnowledgeStore } from "./sqlite-knowledge-store.js";
import type { KnowledgeEntry } from "@teamagent/types";

export interface DualLayerStoreConfig {
  projectDbPath: string;
  userGlobalDbPath: string;
}

/**
 * Q1 决策 C —— 混合双层：
 *   <project>/.teamagent/knowledge.db   ← scope.level=personal
 *   ~/.teamagent/global.db              ← scope.level=global
 *
 * 查询时两层合并。team 作用域留到 Phase 4（现在抛错）。
 */
export class DualLayerStore {
  private readonly project: SqliteKnowledgeStore;
  private readonly global: SqliteKnowledgeStore;

  constructor(cfg: DualLayerStoreConfig) {
    this.project = new SqliteKnowledgeStore(openDb(cfg.projectDbPath));
    this.global = new SqliteKnowledgeStore(openDb(cfg.userGlobalDbPath));
  }

  add(entry: KnowledgeEntry): void {
    switch (entry.scope.level) {
      case "personal":
        this.project.add(entry);
        return;
      case "global":
        this.global.add(entry);
        return;
      case "team":
        throw new Error("team-scoped entries are not supported until Phase 4");
      default:
        throw new Error(`unknown scope level: ${(entry.scope as any).level}`);
    }
  }

  getById(id: string): KnowledgeEntry | undefined {
    return this.project.getById(id) ?? this.global.getById(id);
  }

  findActive(): KnowledgeEntry[] {
    return [...this.project.findActive(), ...this.global.findActive()];
  }

  getAll(): KnowledgeEntry[] {
    return [...this.project.getAll(), ...this.global.getAll()];
  }

  getProjectStore(): SqliteKnowledgeStore {
    return this.project;
  }

  getGlobalStore(): SqliteKnowledgeStore {
    return this.global;
  }

  close(): void {
    this.project.close();
    this.global.close();
  }
}
```

```typescript
import { Parser, Language } from "web-tree-sitter";

let initialized = false;
const langCache = new Map<string, Language>();

/** 首次调用时初始化 web-tree-sitter WASM runtime。幂等。 */
export async function initAstMatcher(): Promise<void> {
  if (initialized) return;
  await Parser.init();
  initialized = true;
}

async function loadLanguage(lang: string): Promise<Language | null> {
  if (langCache.has(lang)) return langCache.get(lang)!;

  // 映射到 WASM 文件路径（tree-sitter-typescript / tree-sitter-python 包里自带）
  const wasmPathMap: Record<string, string> = {
    typescript: "tree-sitter-typescript/tree-sitter-typescript.wasm",
    javascript: "tree-sitter-typescript/tree-sitter-typescript.wasm",
    python: "tree-sitter-python/tree-sitter-python.wasm",
  };

  const wasmPath = wasmPathMap[lang];
  if (!wasmPath) return null;

  try {
    const { createRequire } = await import("node:module");
    const require = createRequire(import.meta.url);
    const fullPath = require.resolve(wasmPath);
    const language = await Language.load(fullPath);
    langCache.set(lang, language);
    return language;
  } catch {
    return null;
  }
}

/**
 * 判断 `code` 中位置 `offset` 处是否落在 comment / string literal 内。
 * 如果落在，这个"命中"就不算真使用，应被过滤。
 *
 * 未知语言 / 加载失败 → 返回 false（保守：当作真命中）。
 */
export async function isInsideCommentOrString(
  code: string,
  offset: number,
  lang: string
): Promise<boolean> {
  if (!initialized) await initAstMatcher();
  const language = await loadLanguage(lang);
  if (!language) return false;

  const parser = new Parser();
  parser.setLanguage(language);
  const tree = parser.parse(code);

  const node = tree.rootNode.descendantForIndex(offset);
  let cur: any = node;
  while (cur) {
    const t = cur.type as string;
    if (
      t === "comment" ||
      t === "line_comment" ||
      t === "block_comment" ||
      t === "string" ||
      t === "string_literal" ||
      t === "string_fragment" ||
      t === "template_string"
    ) {
      tree.delete();
      return true;
    }
    cur = cur.parent;
  }
  tree.delete();
  return false;
}
```

**注意**：`web-tree-sitter` 的 Parser API 2026 年版本是同步 `parse`，但 Language.load 必须 async。测试用 async。

```typescript
import type { PreToolUseHookInput } from "@anthropic-ai/claude-agent-sdk";

export interface PreToolUseDeps {
  matcher: {
    match(input: { tool_name: string; tool_input: unknown }): Promise<{ matched: any[] }>;
  };
  eventLog: { append(e: any): void };
}

export interface PreToolUseResult {
  permissionDecision: "allow" | "deny" | "ask";
  permissionDecisionReason?: string;
  systemMessage?: string; // warn 级别附加给 Claude 的 hint
}

export function createPreToolUseHandler(deps: PreToolUseDeps) {
  return async (input: PreToolUseHookInput): Promise<PreToolUseResult> => {
    const { tool_name, tool_input, tool_use_id } = input;
    const now = new Date().toISOString();

    const { matched } = await deps.matcher.match({ tool_name, tool_input });

    if (matched.length === 0) {
      deps.eventLog.append({
        id: `e-${tool_use_id}-passed`,
        kind: "hook-pre.passed",
        tool_use_id,
        timestamp: now,
        schema_version: 1,
      });
      return { permissionDecision: "allow" };
    }

    // 取 enforcement 最严格的那条
    const byTier = matched.sort((a, b) => severityOrder(b.enforcement) - severityOrder(a.enforcement));
    const top = byTier[0];

    // 构造 structured auto-repair 输出（SP-1 项 5）
    const reason = formatReason(top);

    if (top.enforcement === "block") {
      deps.eventLog.append({
        id: `e-${tool_use_id}-blocked`,
        kind: "hook-pre.blocked",
        knowledge_id: top.id,
        tool_use_id,
        timestamp: now,
        schema_version: 1,
      });
      return { permissionDecision: "deny", permissionDecisionReason: reason };
    }

    // warn / suggest
    deps.eventLog.append({
      id: `e-${tool_use_id}-warned`,
      kind: "hook-pre.warned",
      knowledge_id: top.id,
      tool_use_id,
      timestamp: now,
      schema_version: 1,
    });
    return { permissionDecision: "allow", systemMessage: reason };
  };
}

function severityOrder(e: string): number {
  return { block: 3, warn: 2, suggest: 1, passive: 0 }[e as "block" | "warn"] ?? 0;
}

function formatReason(rule: any): string {
  // 基础 auto-repair 格式；SP-1 项 5 进一步加结构化字段
  return `[${rule.current_tier}] ${rule.trigger}: ${rule.correct_pattern}\n理由: ${rule.reasoning}`;
}
```

```typescript
import type { PostToolUseHookInput } from "@anthropic-ai/claude-agent-sdk";

export interface PostToolUseDeps {
  eventLog: {
    append(e: any): void;
    readLast(n: number): any[];
  };
}

export function createPostToolUseHandler(deps: PostToolUseDeps) {
  return async (input: PostToolUseHookInput): Promise<Record<string, never>> => {
    const { tool_use_id, tool_response } = input;
    const now = new Date().toISOString();
    const success = inferToolSuccess(tool_response);

    // 找本 tool_use_id 对应的 Pre 事件
    const recent = deps.eventLog.readLast(50);
    const preEvents = recent.filter(
      e => e.tool_use_id === tool_use_id && e.kind.startsWith("hook-pre.")
    );

    for (const pre of preEvents) {
      if (!pre.knowledge_id) continue;
      deps.eventLog.append({
        id: `e-post-${tool_use_id}-${pre.knowledge_id}`,
        kind: "hook-post.result",
        knowledge_id: pre.knowledge_id,
        tool_use_id,
        timestamp: now,
        schema_version: 1,
        payload: { success, source_pre_kind: pre.kind },
      });
    }

    return {};
  };
}

export function inferToolSuccess(toolResponse: unknown): boolean {
  if (toolResponse === null || toolResponse === undefined) return true;
  if (typeof toolResponse !== "object") return true;
  const r = toolResponse as Record<string, unknown>;
  if (r.is_error === true) return false;
  if (r.error) return false;
  if (typeof r.exit_code === "number" && r.exit_code !== 0) return false;
  return true;
}
```

```typescript
  const chunks: Buffer[] = [];
  for await (const chunk of process.stdin) chunks.push(chunk as Buffer);
  return JSON.parse(Buffer.concat(chunks).toString("utf-8"));
}

async function main() {
  try {
    const input = await readStdinJson();
    const cwd = normalizeCwd(input.cwd ?? process.cwd());

    const projectDbPath = path.join(cwd, ".teamagent", "knowledge.db");
    const globalDbPath = path.join(os.homedir(), ".teamagent", "global.db");
```
