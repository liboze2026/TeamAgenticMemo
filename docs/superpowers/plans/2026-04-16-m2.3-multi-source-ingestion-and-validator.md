**Goal:** 实现 TeamAgent Phase 2 M2.3 —— 分级 Validator（L0 机械 / L1 Haiku / L2 Sonnet）+ 多源规则摄入（6 源：Detector / /insights / npm audit / PR review / git log / CI）。让"规则入库前有机械门、晋升高 tier 前有 LLM 语义门、数据来源不再仅限于 session 纠正"。

**Architecture:**
- Functional-Core / Imperative-Shell 继续。Validator 是 Port（ L0/L1/L2 三级各自纯函数或注入 LLM 的函数），IO（gh/npm/git）在 adapter 层。
- **3 级 Validator 触发时机不同**：
  - **L0**：规则入库前（`store.add` 前的门闸），**每条都跑**，零 LLM 成本
  - **L1**：规则晋升 stable 时（hysteresis 批准后、实际落地前），走 Haiku
  - **L2**：规则晋升 canonical+ 时，走 Sonnet
  - 因此 Validator 接入点**不是一个**——一处在 ingest-pipeline，另一处在 calibration-pipeline-v2 的 `applyHysteresis` 下游
- **双 pipeline 共存**：原 `extract-pipeline`（Detector 源的 session 纠正）和新增 `ingest-pipeline`（其余 5 源）各自独立入口但**共用 L0 门闸 + store + extractor**。区别在于输入形式：前者从 `ParsedSession` 抽 correction moments 喂 extractor，后者直接吃 `ExtractionInput[]`。不合并成单 pipeline 是因为 session 解析与 raw 材料解析职责差距大，强合并会让 detector deps 成为 ingest 的死权重。
- **多源摄入采用 source-agnostic pipeline**：每个源只负责把原始材料转成 `ExtractionInput[]`；后续 extractor → L0 → store 共用一条通路
- **6 源分两档**：
  - 自动档（4 源，高 SNR）：Detector（已有）/ /insights / npm audit / PR review
  - 半自动档（2 源，低 SNR）：git log hotspot / CI failure log —— 输出 candidate 列表让用户勾选
- GitHub/CI 依赖 `gh` CLI 桥接（项目已习惯用 `gh`），不引入 Octokit，保持 local-first

**Tech Stack:** TypeScript、Vitest、zod、pnpm workspace、`gh` CLI（child_process 包装，adapter 层）、`npm audit --json`。

**参考文档：**
- 设计源：`docs/superpowers/specs/2026-04-15-phase2-design-v2.md` §5.3（Validator 分级）+ §5.5（多源摄入）+ §十三（M2.3 拆分）
- 依赖的已落地基础：`docs/superpowers/plans/2026-04-16-m2.2-calibrator-v2.md`（Calibrator v2 + observations + tier/demerit）
- 项目开发约定：`CLAUDE.md`（Functional Core / Port 先写契约 / 小 commit）
- Backlog 中相关条目：`docs/specs/2026-04-15-phase2-backlog.md` §B1（scope.paths 自动推断）/ §D（规则质量审查）

**关键决策（已确认 / 可挑战）：**
1. `ExtractionInput.kind` 从 3 扩到 7：`correction | success | rule-text | insights | npm-audit | pr-review | git-hotspot | ci-failure`。prompt 按 kind 分派不同 header。
2. Validator Port 是**3 个方法的单接口**（`validateLevel0/1/2`），不是 3 个 port。让实现只需一个导出对象，调用方按需分派。
3. 半自动 CLI 产出的 candidate 文件是**可编辑的 TOML/Markdown**——用户勾选后直接 `teamagent ingest --from-candidates <file>` 吃回去。不做交互式 TUI（M2.3 范围太重）。
4. L1/L2 失败的规则**不降级已有 tier**，只阻止**继续晋升**（候选 tier 回落当前 tier）。避免 Haiku 误判反噬历史积累。
5. PR / CI 源**通过 gh CLI** 实现（已是项目依赖环境），不引入 HTTP client 或 Octokit。

```typescript
import type { KnowledgeEntry } from "@teamagent/types";

/** L0 机械检查的结果。 */
export interface ValidationL0Result {
  ok: boolean;
  failed_checks: string[];           // 具体失败项，如 "wrong_pattern_not_in_source"
  notes?: string;
}

/** L1/L2 LLM 检查的结果。 */
export interface ValidationLLMResult {
  ok: boolean;
  confidence: number;                // LLM 自评 0-1
  reason: string;                    // 一两句人话解释
  conflicts_with?: string[];         // 与哪些规则 id 矛盾
}

/** L0 输入（无 LLM，纯机械）。 */
export interface ValidateL0Input {
  entry: Partial<KnowledgeEntry>;
  /** 规则来源的源文（diff / transcript / 原 PR 评论等），L0 用它验证 wrong_pattern 确实存在。 */
  sourceText: string;
  /** 项目已有规则（用于 trigger 字面冲突检查）。 */
  existingRules: Pick<KnowledgeEntry, "id" | "trigger" | "wrong_pattern">[];
  /** 项目探测到的 stack（file_types 一致性检查）。 */
  projectStack: string[];            // e.g. ["ts", "tsx", "md"]
  /** 工作区根目录，用于校验 import_path / scope.paths 存在性（L0 只看字符串合法性，不真的 stat）。 */
  workspaceRoot?: string;
}

/** L1 输入（Haiku）。 */
export interface ValidateL1Input {
  entry: KnowledgeEntry;
  /** 近似规则 top-k（embedding 召回），用于矛盾检测。 */
  similarRules: KnowledgeEntry[];
}

/** L2 输入（Sonnet）。 */
export interface ValidateL2Input {
  entry: KnowledgeEntry;
  /** 最近 20 次命中的真实 tool_input 样本，用于过拟合检测。 */
  recentHits: { tool_input: unknown; timestamp: string }[];
  /** 项目已有 stable+ 规则。 */
  existingSeniorRules: KnowledgeEntry[];
}

/** Validator 统一接口。三级之中任意级都可不实现（返回 { ok: true, ... }）以便降级使用。 */
export interface Validator {
  validateLevel0(input: ValidateL0Input): ValidationL0Result;
  validateLevel1(
    input: ValidateL1Input,
    callLLM: (prompt: string) => Promise<string>,
  ): Promise<ValidationLLMResult>;
  validateLevel2(
    input: ValidateL2Input,
    callLLM: (prompt: string) => Promise<string>,
  ): Promise<ValidationLLMResult>;
}
```

```typescript
import { describe, it, expect } from "vitest";
import type { Validator } from "../validator.js";
import type { KnowledgeEntry } from "@teamagent/types";

/**
 * Validator 契约不变量：
 *
 * L0:
 *  1. 5 项机械检查全绿 → ok=true, failed_checks 为空
 *  2. wrong_pattern 完全不在 sourceText 里 → ok=false, failed_checks 含 "wrong_pattern_not_in_source"
 *  3. scope.paths 空 + type=avoidance → ok=false, failed_checks 含 "scope_paths_empty"
 *  4. existingRules 里有相同 trigger → ok=false, failed_checks 含 "trigger_collision"
 *  5. 纯函数：相同输入 → 相同输出
 *
 * L1/L2:
 *  6. confidence ∈ [0, 1]
 *  7. ok=false 时必带 reason
 *  8. LLM 返回 garbage → ok=false（不抛异常）
 */
export function runValidatorContract(make: () => Validator): void {
  describe("Validator contract", () => {
    const baseEntry: KnowledgeEntry = {
      id: "r1",
      scope: { level: "team", paths: ["src/**"] },
      category: "E",
      tags: [],
      type: "avoidance",
      nature: "subjective",
      trigger: "use-fetch-not-axios",
      wrong_pattern: "axios",
      correct_pattern: "fetch",
      reasoning: "r",
      confidence: 0.5,
      enforcement: "suggest",
      status: "active",
      hit_count: 0,
      success_count: 0,
      override_count: 0,
      evidence: { success_sessions: 0, success_users: 0, correction_sessions: 0 },
      created_at: "2026-04-16T00:00:00Z",
      last_hit_at: "",
      last_validated_at: "2026-04-16T00:00:00Z",
      source: "accumulated",
      conflict_with: [],
      current_tier: "experimental",
      max_tier_ever: "experimental",
      tier_entered_at: "2026-04-16T00:00:00Z",
      demerit: 0,
      demerit_last_updated: "",
      resurrect_count: 0,
    };

    describe("L0", () => {
      it("all 5 mechanical checks pass → ok=true", () => {
        const r = make().validateLevel0({
          entry: baseEntry,
          sourceText: "import axios from 'axios';",
          existingRules: [],
          projectStack: ["ts"],
        });
        expect(r.ok).toBe(true);
        expect(r.failed_checks).toEqual([]);
      });

      it("wrong_pattern absent in source → fail", () => {
        const r = make().validateLevel0({
          entry: baseEntry,
          sourceText: "console.log('hi');",
          existingRules: [],
          projectStack: ["ts"],
        });
        expect(r.ok).toBe(false);
        expect(r.failed_checks).toContain("wrong_pattern_not_in_source");
      });

      it("avoidance with empty scope.paths → fail", () => {
        const r = make().validateLevel0({
          entry: { ...baseEntry, scope: { level: "team", paths: [] } },
          sourceText: "import axios from 'axios';",
          existingRules: [],
          projectStack: ["ts"],
        });
        expect(r.ok).toBe(false);
        expect(r.failed_checks).toContain("scope_paths_empty");
      });

      it("trigger collision with existing rule → fail", () => {
        const r = make().validateLevel0({
          entry: baseEntry,
          sourceText: "import axios from 'axios';",
          existingRules: [
            { id: "r-old", trigger: "use-fetch-not-axios", wrong_pattern: "axios" },
          ],
          projectStack: ["ts"],
        });
        expect(r.ok).toBe(false);
        expect(r.failed_checks).toContain("trigger_collision");
      });

      it("is pure (same input → same output)", () => {
        const input = {
          entry: baseEntry,
          sourceText: "import axios from 'axios';",
          existingRules: [],
          projectStack: ["ts"],
        };
        const a = make().validateLevel0(input);
        const b = make().validateLevel0(input);
        expect(a).toEqual(b);
      });
    });

    describe("L1/L2", () => {
      it("confidence within [0,1]", async () => {
        const stubLLM = async () =>
          JSON.stringify({ ok: true, confidence: 0.8, reason: "fine" });
        const r = await make().validateLevel1(
          { entry: baseEntry, similarRules: [] },
          stubLLM,
        );
        expect(r.confidence).toBeGreaterThanOrEqual(0);
        expect(r.confidence).toBeLessThanOrEqual(1);
      });

      it("ok=false requires reason", async () => {
        const stubLLM = async () =>
          JSON.stringify({ ok: false, confidence: 0.2, reason: "too broad" });
        const r = await make().validateLevel1(
          { entry: baseEntry, similarRules: [] },
          stubLLM,
        );
        if (!r.ok) expect(r.reason.length).toBeGreaterThan(0);
      });

      it("garbage LLM response → ok=false, no throw", async () => {
        const stubLLM = async () => "lol not json";
        const r = await make().validateLevel1(
          { entry: baseEntry, similarRules: [] },
          stubLLM,
        );
        expect(r.ok).toBe(false);
      });
    });
  });
}
```

```typescript
import type { ValidateL0Input, ValidationL0Result } from "@teamagent/ports";

/** L0 机械检查。纯函数，无 IO。见 design §5.3。 */
export function validateLevel0(input: ValidateL0Input): ValidationL0Result {
  const failed: string[] = [];
  const { entry, sourceText, existingRules, projectStack } = input;

  // 1. wrong_pattern 在源文里真存在
  if (entry.type === "avoidance" && entry.wrong_pattern) {
    const patterns = entry.wrong_pattern.split("|").map((s) => s.trim()).filter(Boolean);
    const hit = patterns.some((p) => sourceText.includes(p));
    if (!hit) failed.push("wrong_pattern_not_in_source");
  }

  // 2. import_path 字符串合法
  const importPath = (entry as any).correct_pattern_import_path;
  if (importPath !== undefined && !/^[@a-zA-Z0-9_\-./]+$/.test(importPath)) {
    failed.push("invalid_import_path_format");
  }

  // 3. scope.file_types 与 stack 一致
  const fileTypes = entry.scope?.file_types;
  if (fileTypes && fileTypes.length > 0 && projectStack.length > 0) {
    const norm = fileTypes.map((s) => s.replace(/^\*?\.?/, ""));
    const overlap = norm.some((t) => projectStack.includes(t));
    if (!overlap) failed.push("file_types_stack_mismatch");
  }

  // 4. trigger 字面冲突
  if (entry.trigger) {
    const exists = existingRules.some(
      (r) => r.id !== entry.id && r.trigger === entry.trigger,
    );
    if (exists) failed.push("trigger_collision");
  }

  // 5. avoidance 规则 scope.paths 不为空 且路径格式合法
  if (entry.type === "avoidance") {
    const paths = entry.scope?.paths ?? [];
    if (paths.length === 0) {
      failed.push("scope_paths_empty");
    } else {
      const illegal = paths.some((p) => typeof p !== "string" || p.length === 0);
      if (illegal) failed.push("scope_paths_malformed");
    }
  }

  return {
    ok: failed.length === 0,
    failed_checks: failed,
    notes: failed.length ? `L0 failed: ${failed.join(", ")}` : undefined,
  };
}
```

```typescript
import type {
  AttributionBus,
  ExtractionInput,
  KnowledgeExtractor,
  KnowledgeStore,
  Validator,
} from "@teamagent/ports";
import type { KnowledgeEntry, Scope } from "@teamagent/types";
import { computeEnforcement } from "@teamagent/types";

export interface IngestPipelineDeps {
  inputs: ExtractionInput[];
  extractor: KnowledgeExtractor;
  callLLM: (prompt: string) => Promise<string>;
  validator: Validator;
  store: KnowledgeStore;
  bus?: AttributionBus;
  scope: Scope;
  source: KnowledgeEntry["source"];
  projectStack: string[];
  now: () => Date;
  idGen: () => string;
  dryRun?: boolean;
}

export interface IngestPipelineResult {
  scanned: number;
  accepted: KnowledgeEntry[];
  rejected: { partial: Partial<KnowledgeEntry>; reasons: string[] }[];
  skipped: number;   // LLM 返 null
  failed: number;
}

export async function runIngestPipeline(
  deps: IngestPipelineDeps,
): Promise<IngestPipelineResult> {
  const accepted: KnowledgeEntry[] = [];
  const rejected: { partial: Partial<KnowledgeEntry>; reasons: string[] }[] = [];
  let skipped = 0;
  let failed = 0;

  for (const input of deps.inputs) {
    let partial: Partial<KnowledgeEntry> | null = null;
    try {
      partial = await deps.extractor.extract(input, deps.callLLM);
    } catch {
      failed += 1;
      continue;
    }
    if (!partial) {
      skipped += 1;
      continue;
    }

    const entry: KnowledgeEntry = completeEntry(partial, deps);

    const l0 = deps.validator.validateLevel0({
      entry,
      sourceText: input.context,
      existingRules: deps.store.getAll().map(({ id, trigger, wrong_pattern }) => ({
        id, trigger, wrong_pattern,
      })),
      projectStack: deps.projectStack,
    });

    if (!l0.ok) {
      rejected.push({ partial: entry, reasons: l0.failed_checks });
      deps.bus?.emit({
        source: "ingest",
        action: "rejected_l0",
        target: { id: entry.id },
        severity: "info",
        userFacingValue: `rejected: ${l0.failed_checks.join(",")}`,
        timestamp: deps.now().toISOString(),
      });
      continue;
    }

    if (!deps.dryRun) deps.store.add(entry);
    accepted.push(entry);
    deps.bus?.emit({
      source: "ingest",
      action: "accepted",
      target: { id: entry.id },
      severity: "info",
      userFacingValue: `accepted: ${entry.trigger}`,
      timestamp: deps.now().toISOString(),
    });
  }

  return { scanned: deps.inputs.length, accepted, rejected, skipped, failed };
}

function completeEntry(
  partial: Partial<KnowledgeEntry>,
  deps: IngestPipelineDeps,
): KnowledgeEntry {
  const nowIso = deps.now().toISOString();
  const confidence = 0;
  return {
    id: partial.id ?? deps.idGen(),
    scope: partial.scope ?? deps.scope,
    category: partial.category ?? "E",
    tags: partial.tags ?? [],
    type: partial.type ?? "avoidance",
    nature: partial.nature ?? "subjective",
    trigger: partial.trigger ?? "",
    wrong_pattern: partial.wrong_pattern,
    correct_pattern: partial.correct_pattern ?? "",
    reasoning: partial.reasoning ?? "",
    confidence,
    enforcement: partial.enforcement ?? computeEnforcement(confidence),
    status: "active",
    hit_count: 0,
    success_count: 0,
    override_count: 0,
    evidence: { success_sessions: 0, success_users: 0, correction_sessions: 0 },
    created_at: nowIso,
    last_hit_at: "",
    last_validated_at: nowIso,
    source: deps.source,
    conflict_with: [],
    current_tier: "experimental",
    max_tier_ever: "experimental",
    tier_entered_at: nowIso,
    demerit: 0,
    demerit_last_updated: "",
    resurrect_count: 0,
  };
}
```

```typescript
export function parseNpmAudit(raw: string): ExtractionInput[] {
  const data = JSON.parse(raw);
  const vulns = data.vulnerabilities ?? {};
  const out: ExtractionInput[] = [];
  for (const [pkg, v] of Object.entries<any>(vulns)) {

      kind: "npm-audit",
      context: `[severity=${v.severity}] ${pkg}: ${v.title ?? ""} (${v.url ?? "no url"})`,
      weight: v.severity === "critical" ? 1.0 : 0.8,

}
```

```typescript
export interface GitHotspot {
  path: string;
  change_count: number;
}

export function parseGitHotspots(
  logOutput: string,
  opts: { threshold: number },
): GitHotspot[] {
  const counts = new Map<string, number>();
  for (const line of logOutput.split("\n")) {
    const m = line.match(/^\s*\d+\s+\d+\s+(.+)$/);
    if (!m) continue;
    counts.set(m[1], (counts.get(m[1]) ?? 0) + 1);
  }
  return [...counts.entries()]
    .filter(([, c]) => c >= opts.threshold)
    .map(([path, c]) => ({ path, change_count: c }))
    .sort((a, b) => b.change_count - a.change_count);
}

export function formatCandidateMd(hotspots: GitHotspot[]): string {
  return [
    "# TeamAgent ingest candidates (git hotspot)",
    "",
    "勾选后运行:",
    "  teamagent ingest --from-candidates <this-file>",
    "",
    ...hotspots.map((h, i) =>
      `- [ ] Candidate ${i + 1}: \`${h.path}\` 被修改 ${h.change_count} 次`,
    ),
  ].join("\n");
}
```

```typescript
import type {
  ValidateL1Input,
  ValidationLLMResult,
} from "@teamagent/ports";

export async function validateLevel1(
  input: ValidateL1Input,
  callLLM: (prompt: string) => Promise<string>,
): Promise<ValidationLLMResult> {
  const prompt = buildL1Prompt(input);
  let raw: string;
  try {
    raw = await callLLM(prompt);
  } catch (e) {
    return { ok: false, confidence: 0, reason: `llm_error: ${String(e)}` };
  }
  return parseLLMValidation(raw, input);
}

function buildL1Prompt(input: ValidateL1Input): string {
  return [
    "你是规则质量审查官（L1）。请判断这条规则是否足够 specific 且无矛盾。",
    "",
    "【待审规则】",
    JSON.stringify({
      trigger: input.entry.trigger,
      wrong_pattern: input.entry.wrong_pattern,
      correct_pattern: input.entry.correct_pattern,
      reasoning: input.entry.reasoning,
      scope: input.entry.scope,
    }, null, 2),
    "",
    "【近邻规则（embedding 召回）】",
    input.similarRules.map((r) => `- ${r.id}: ${r.trigger}`).join("\n") || "(无)",
    "",
    "返回严格 JSON:",
    '{"ok": true|false, "confidence": 0-1, "reason": "...", "conflicts_with": ["id1"]}',
  ].join("\n");
}

function parseLLMValidation(
  raw: string,
  input: ValidateL1Input,
): ValidationLLMResult {
  try {
    const obj = JSON.parse(raw.trim().replace(/^```json\s*|\s*```$/g, ""));
    const ok = typeof obj.ok === "boolean" ? obj.ok : false;
    const confidence = typeof obj.confidence === "number"
      ? Math.max(0, Math.min(1, obj.confidence))
      : 0;
    const reason = typeof obj.reason === "string" ? obj.reason : "no_reason";
    return {
      ok,
      confidence,
      reason,
      conflicts_with: Array.isArray(obj.conflicts_with) ? obj.conflicts_with : undefined,
    };
  } catch {
    return { ok: false, confidence: 0, reason: "llm_response_unparseable" };
  }
}
```

**(b) 新增内部辅助 `overrideTier`（在本文件内，不公开）：**

```typescript
function overrideTier(
  result: CalibrationResultV2,
  revertTo: Tier,
  reason: string,
): CalibrationResultV2 {
  return {
    ...result,
    tier_after: revertTo,
    tier_transition: null,
    reason_for_no_transition: reason,
    delta_breakdown: [
      ...result.delta_breakdown,
      { type: "tier_transition", note: `reverted to ${revertTo}: ${reason}` },
    ],
  };
}

function isPromotion(from: Tier, to: Tier): boolean {
  const rank = (t: Tier) => ["experimental","probation","stable","canonical","enforced"].indexOf(t);
  return rank(to) > rank(from);
}
```

```typescript
let result = deps.calibrator.calibrate(entry, { events: evtForEntry, observations: obsForEntry, now });

if (
  deps.validator && deps.callLLM &&
  result.tier_transition && isPromotion(result.tier_before, result.tier_after)
) {
  // L1 门（晋升到 stable / canonical / enforced 都要跑；canonical+ 还要额外跑 L2）
  const needsL1 = ["stable", "canonical", "enforced"].includes(result.tier_after);
  if (needsL1) {
    const similar = await (deps.similarityFinder?.(entry) ?? Promise.resolve([]));
    const l1 = await deps.validator.validateLevel1({ entry, similarRules: similar }, deps.callLLM);
    if (!l1.ok) {
      result = overrideTier(result, entry.current_tier as Tier, `l1_blocked: ${l1.reason}`);
      deps.bus?.emit({
        source: "validator",
        action: "blocked_promotion",
        target: { id: entry.id },
        severity: "info",
        userFacingValue: `L1 blocked ${entry.current_tier} → ${result.tier_before}: ${l1.reason}`,
        timestamp: now.toISOString(),
      });
    }
  }
  // L2 门（只在晋升 canonical / enforced 时跑）
  const needsL2 = ["canonical", "enforced"].includes(result.tier_after) && result.tier_transition;
  if (needsL2) {
    const recentHits = deps.events
      .filter((e) => e.knowledge_id === entry.id && e.kind === "hook-pre.matched")
      .slice(-20)
      .map((e) => ({ tool_input: (e.payload as any)?.tool_input ?? null, timestamp: e.timestamp }));
    const seniors = deps.store.getAll().filter((r) =>
      ["canonical", "enforced"].includes(r.current_tier) && r.id !== entry.id,
    );
    const l2 = await deps.validator.validateLevel2({ entry, recentHits, existingSeniorRules: seniors }, deps.callLLM);
    if (!l2.ok) {
      result = overrideTier(result, entry.current_tier as Tier, `l2_blocked: ${l2.reason}`);
      deps.bus?.emit({
        source: "validator",
        action: "blocked_promotion",
        target: { id: entry.id },
        severity: "info",
        userFacingValue: `L2 blocked: ${l2.reason}`,
        timestamp: now.toISOString(),
      });
    }
  }
}
```
